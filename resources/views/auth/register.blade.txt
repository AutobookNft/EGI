        document.addEventListener('DOMContentLoaded', function() {
            // User type selection enhancement
            const userTypeInputs = document.querySelectorAll('input[name="user_type"]');
            const cards = document.querySelectorAll('.consent-card');

            userTypeInputs.forEach(input => {
                input.addEventListener('change', function() {
                    // Remove selection from all cards
                    cards.forEach(card => {
                        card.classList.remove('ring-2', 'ring-oro-fiorentino', 'bg-oro-fiorentino/5');
                    });

                    // Add selection to current card
                    if (this.checked) {
                        this.closest('.consent-card').classList.add('ring-2', 'ring-oro-fiorentino', 'bg-oro-fiorentino/5');
                    }
                });

                // Check on page load if already selected
                if (input.checked) {
                    input.closest('.consent-card').classList.add('ring-2', 'ring-oro-fiorentino', 'bg-oro-fiorentino/5');
                }
            });

            // Password strength indicator
            const passwordInput = document.getElementById('password');
            const passwordHelp = document.getElementById('password-help');

            if (passwordInput && passwordHelp) {
                passwordInput.addEventListener('input', function() {
                    const value = this.value;
                    const strength = calculatePasswordStrength(value);
                    updatePasswordHelp(passwordHelp, strength);
                });
            }

            function calculatePasswordStrength(password) {
                let strength = 0;
                if (password.length >= 8) strength++;
                if (/[a-z]/.test(password)) strength++;
                if (/[A-Z]/.test(password)) strength++;
                if (/[0-9]/.test(password)) strength++;
                if (/[^A-Za-z0-9]/.test(password)) strength++;
                return strength;
            }

            function updatePasswordHelp(element, strength) {
                const messages = [
                    'Password troppo debole',
                    'Password debole',
                    'Password accettabile',
                    'Password buona',
                    'Password forte',
                    'Password eccellente'
                ];

                const colors = [
                    'text-rosso-urgenza',
                    'text-rosso-urgenza',
                    'text-yellow-600',
                    'text-verde-rinascita',
                    'text-verde-rinascita',
                    'text-verde-rinascita'
                ];

                element.className = `mt-1 text-xs ${colors[strength]}`;
                element.textContent = messages[strength] || messages[0];
            }

            // Form validation enhancement
            const form = document.querySelector('form');
            if (form) {
                form.addEventListener('submit', function(e) {
                    // Check required consents
                    const requiredConsents = document.querySelectorAll('input[type="checkbox"][required]');
                    let hasErrors = false;

                    requiredConsents.forEach(input => {
                        if (!input.checked) {
                            hasErrors = true;
                            input.closest('.consent-card').classList.add('ring-2', 'ring-rosso-urgenza');
                        } else {
                            input.closest('.consent-card').classList.remove('ring-2', 'ring-rosso-urgenza');
                        }
                    });

                    if (hasErrors) {
                        e.preventDefault();
                        document.querySelector('.consent-card.ring-rosso-urgenza').scrollIntoView({
                            behavior: 'smooth',
                            block: 'center'
                        });
                    }
                });
            }
        });
